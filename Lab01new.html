## Fancy graphs: contour, image, persp (3D)
f=outer(x,y,function(x,y)cos(y)/(1+x^2))
contour(x,y,f)
contour(x,y,f,nlevels=45,add=T)

fa=(f-t(f))/2 # matrix transpose
contour(x,y,fa,nlevels=15)

image(x,y,fa)

persp(x,y,fa)
